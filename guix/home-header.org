;; Put logs into XDG_LOG_HOME/#$name.log
;; Or $HOME/.local/var/log/$#name.log
(define (home-log name)
        #~(string-append (format #f "~a"
                                 (or (getenv "XDG_LOG_HOME")
                                     (format #f "~a/.local/var/log"
                                             (getenv "HOME"))))
                         "/" #$name ".log"))

(define %home-services '())
